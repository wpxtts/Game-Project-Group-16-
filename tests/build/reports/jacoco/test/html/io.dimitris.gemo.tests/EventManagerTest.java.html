package io.dimitris.gemo.tests;
import com.skloch.game.EventManager;
import com.skloch.game.GameScreen;
import com.skloch.game.HustleGame;
import org.junit.Test;
import org.junit.runner.RunWith;

import static org.junit.Assert.assertEquals;
import static org.junit.Assert.assertFalse;
@RunWith(GdxTestRunner.class)
public class EventManagerTest {
    @Test
    public void testPiazzaEvent(){
        String[] args = new String[]{"piazza","1"};
        HustleGame hustleGame = new HustleGame(800, 800);
        GameScreen gameScreen = new GameScreen(hustleGame,0);
        EventManager eventManager = new EventManager(gameScreen);
        int energy = gameScreen.getEnergy();
        eventManager.piazzaEvent(args);
        assertEquals(energy-10, gameScreen.getEnergy());
    }
}
